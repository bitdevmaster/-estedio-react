## Config helpers

These helpers are exported from the root package and re-exported by every adapter package.

### initializeLibConfig

```ts
function initializeLibConfig(config: Partial<LibraryConfig>): void;
```

### getLibConfig

```ts
function getLibConfig(
  overrides?: Partial<LibraryConfig>,
  env?: Record<string, string | undefined>,
): LibraryConfig;
```

### dataProvider

```ts
function dataProvider(config: DataProviderConfig): Partial<LibraryConfig>;
```

### resetLibConfig

```ts
function resetLibConfig(): void;
```

## Core types

### LibraryConfig

```ts
interface LibraryConfig {
  nodeEnv: string;
  apiBaseUrl: string;
  refreshEndpoint: string;
  accessTokenKey: string;
  refreshTokenKey: string;
  storageSecret: string;
}
```

### DataProviderConfig

```ts
interface DataProviderConfig {
  access_token_key?: string;
  refresh_token_key?: string;
  access_token?: string;
  refresh_token?: string;
  base_url?: string;
  refresh_endpoint?: string;
  storage_secret?: string;
}
```

### ResourceConfig

```ts
interface ResourceConfig<TPayload = unknown, TResponse = unknown> {
  endpoint: string;
  method: HttpMethod;
  authenticated?: boolean;
  sagaEffect?: SagaEffect;
  meta?: ResourceConfigMeta;
  payload?: TPayload;
  response?: TResponse;
}
```

### ResourceConfigMeta

```ts
interface ResourceConfigMeta {
  persist?: boolean;
  persistKey?: string;
  isAuthResource?: boolean;
}
```

### ResourceState

```ts
interface ResourceState<T = unknown> {
  result: T | null;
  error: string | null;
  loading: boolean;
}
```

### NormalizedApiError

```ts
interface NormalizedApiError {
  status: number;
  code: string;
  ray_id: string;
  message: string;
}
```

## Error utilities

### normalizeError

```ts
function normalizeError(error: unknown): NormalizedApiError;
```

Converts arbitrary thrown values into the shared normalized error shape.

### isNormalizedApiError

```ts
function isNormalizedApiError(value: unknown): value is NormalizedApiError;
```

## Ports

### IApiPort

```ts
interface IApiPort {
  execute<T = unknown>(resource: string, payload?: unknown): Promise<T>;
}
```

### IStoragePort

```ts
interface IStoragePort {
  getAsync(key: string): Promise<string | null>;
  setAsync(key: string, value: string): Promise<void>;
  remove(key: string): void;
  clear(): void;
}
```

## HTTP and storage exports

### ResourceAdapter

```ts
class ResourceAdapter implements IApiPort {
  constructor(registry: Record<string, ResourceConfig>);
  execute<T>(resource: string, payload?: unknown): Promise<T>;
}
```

### apiClient

Shared Axios-backed client with auth-header injection and refresh-token retry behavior.

### tokenManager

Default token manager instance used by the adapters.

### createTokenManager

```ts
function createTokenManager(
  storage: IStoragePort,
  config: TokenManagerConfig | (() => TokenManagerConfig),
): TokenManager;
```

### localStorageAdapter

Encrypted `window.localStorage` adapter.

### sessionStorageAdapter

Encrypted `window.sessionStorage` adapter.

### initialResourceState

```ts
const initialResourceState: ResourceState = {
  result: null,
  error: null,
  loading: false,
};
```
