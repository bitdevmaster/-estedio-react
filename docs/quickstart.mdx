## 1. Install

Install the core package and `axios`, which is always required:

```bash
npm install @estedio/react axios
```

Then install the peer dependencies for the adapter you want to use:

<Tabs>
  <Tab title="Redux">
    ```bash npm install @reduxjs/toolkit react-redux redux-saga ```
  </Tab>
  <Tab title="Zustand">```bash npm install zustand ```</Tab>
  <Tab title="TanStack Query">
    ```bash npm install @tanstack/react-query ```
  </Tab>
</Tabs>

## 2. Configure

Call `initializeLibConfig` once at application startup, before any resource is fetched:

```ts
// app/config.ts
import { initializeLibConfig } from "@estedio/react/redux"; // or /zustand or /tanstack-query

initializeLibConfig({
  apiBaseUrl: "https://api.example.com",
  refreshEndpoint: "/auth/refresh",
  accessTokenKey: "access_token",
  refreshTokenKey: "refresh_token",
  storageSecret: "replace-this-in-production",
});
```

<Tip>
  You can also pass values via environment variables — the library reads
  `ESTEDIO_API_BASE_URL` and `NEXT_PUBLIC_API_BASE_URL` automatically. See
  [Configuration](/configuration) for the full list.
</Tip>

## 3. Define resources

Create a file that defines your resource type map and config objects:

```ts
// resources.ts
import type { ResourceConfig } from "@estedio/react/redux";

type User = { id: number; name: string };
type CreateUserDto = { name: string; email: string };

export interface ResourceTypeMap {
  GetUsers: { payload: void; response: User[] };
  GetUserById: { payload: { userId: number }; response: User };
  CreateUser: { payload: CreateUserDto; response: User };
}

export const resources = {
  GetUsers: {
    endpoint: "/users",
    method: "GET",
    authenticated: true,
    payload: {} as void,
    response: {} as User[],
  },
  GetUserById: {
    endpoint: "/users/:userId",
    method: "GET",
    authenticated: true,
    payload: {} as { userId: number },
    response: {} as User,
  },
  CreateUser: {
    endpoint: "/users",
    method: "POST",
    authenticated: true,
    payload: {} as CreateUserDto,
    response: {} as User,
  },
} satisfies Record<keyof ResourceTypeMap, ResourceConfig>;
```

<Note>
  The `:userId` in the endpoint is automatically replaced with the value from
  the payload at runtime. See [HTTP Client — Path
  parameters](/core/http-client#path-parameters) for details.
</Note>

## 4. Set up your adapter

<Tabs>
  <Tab title="Redux">
    **Create the store:**
    ```ts
    // store.ts
    import { createResources, localStorageAdapter } from "@estedio/react/redux";
    import { resources } from "./resources";

    export const { makeStore, slices } = createResources(resources, {
      storageByResource: {
        GetUsers: localStorageAdapter,
      },
    });
    ```

    **Add the provider:**
    ```tsx
    // app/providers.tsx
    "use client";
    import { StoreProvider } from "@estedio/react/redux";
    import { makeStore } from "./store";

    export function AppProvider({ children }: { children: React.ReactNode }) {
      return (
        <StoreProvider makeStore={makeStore} provider={{ apiBaseUrl: "https://api.example.com" }}>
          {children}
        </StoreProvider>
      );
    }
    ```

    **Create typed hooks:**
    ```ts
    // hooks.ts
    "use client";
    import { createResourceHooks } from "@estedio/react/redux";
    import { slices } from "./store";
    import type { ResourceTypeMap } from "./resources";

    export const { useResourceDispatch, useResourceSelector } =
      createResourceHooks<ResourceTypeMap>(slices);
    ```

  </Tab>
  <Tab title="Zustand">
    ```ts
    // store.ts
    "use client";
    import { createResources, createResourceHooks } from "@estedio/react/zustand";
    import { resources } from "./resources";
    import type { ResourceTypeMap } from "./resources";

    const { stores } = createResources<ResourceTypeMap>(resources);

    export const { useResourceDispatch, useResourceSelector } =
      createResourceHooks<ResourceTypeMap>(stores);
    ```

  </Tab>
  <Tab title="TanStack Query">
    ```tsx
    // app/providers.tsx
    "use client";
    import { QueryClient, QueryClientProvider } from "@tanstack/react-query";
    import { createResources } from "@estedio/react/tanstack-query";
    import { resources } from "./resources";
    import type { ResourceTypeMap } from "./resources";

    const queryClient = new QueryClient();

    export const { useResourceQuery, useResourceMutation } =
      createResources<ResourceTypeMap>(resources);

    export function AppProvider({ children }: { children: React.ReactNode }) {
      return (
        <QueryClientProvider client={queryClient}>
          {children}
        </QueryClientProvider>
      );
    }
    ```

  </Tab>
</Tabs>

## 5. Fetch data in a component

<Tabs>
  <Tab title="Redux / Zustand">
    ```tsx
    "use client";
    import { useResourceDispatch, useResourceSelector } from "./hooks";

    export function UserList() {
      const dispatch = useResourceDispatch();
      const { result, loading, error } = useResourceSelector("GetUsers");

      if (loading) return <p>Loading...</p>;
      if (error) return <p>Error: {error}</p>;

      return (
        <>
          <button onClick={() => dispatch({ resource: "GetUsers", force: false })}>
            Fetch users
          </button>
          <ul>
            {result?.map((user) => <li key={user.id}>{user.name}</li>)}
          </ul>
        </>
      );
    }
    ```

  </Tab>
  <Tab title="TanStack Query">
    ```tsx
    "use client";
    import { useResourceQuery, useResourceMutation } from "./providers";

    export function UserList() {
      const { result, loading, error } = useResourceQuery({
        resource: "GetUsers",
        force: false,
      });
      const createUser = useResourceMutation("CreateUser");

      if (loading) return <p>Loading...</p>;
      if (error) return <p>Error: {error}</p>;

      return (
        <>
          <button onClick={() => createUser.dispatch({ name: "Alice", email: "alice@example.com" })}>
            Create user
          </button>
          <ul>
            {result?.map((user) => <li key={user.id}>{user.name}</li>)}
          </ul>
        </>
      );
    }
    ```

  </Tab>
</Tabs>
